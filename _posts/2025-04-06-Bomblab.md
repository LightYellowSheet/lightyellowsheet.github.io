函数fun7接受两个参数rdi rsi，写以下伪代码
猜测rdi是指针
```C
int fun7(int* rdi, int rsi) {
	if(rdi == 0) {
		return -1;
	}
	int edx = *rdi; //第一次的值是36
	if(edx <= rsi) {
		int rax = 0;
		if(edx == rsi) {
			return 0;
		}
		fun7(rdi+16, rsi);
		eax = 1 + 2*eax;
		return eax;
	}
	fun7(rdi+8, rsi);
	eax += eax;
	return eax;
}
```
